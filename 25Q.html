import matplotlib.pyplot as plt
def fibuptoN(n):
    if n == 1:
        return [1]
    elif n == 2:
        return [1, 1]
    f1 = 1
    f2 = 1
    lst = [1, 1]
    for i in range(n - 2):
        temp = f2
        f2 += f1
        f1 = temp
        lst.append(f2)
    return lst
n = int(input())

ypt = fibuptoN(n)
